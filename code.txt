int factorial(int n) {
    int ans = 1;
    while (n > 0) {
        ans = ans * n;
        n = n - 1;
    };
    return ans;
};

int get_answer(int n) {
    return factorial(n);
};

    if (52 * 4) {

    } else {

    };

};
